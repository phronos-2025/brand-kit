// Usage

Initialize the logo with JavaScript:

// Basic usage
new PhronosLogo('container-id');

// With options - SCALING NOW WORKS PROPERLY!
new PhronosLogo('container-id', {
    theme: 'dark',      // 'dark' or 'light'
    showText: true,     // show ".p.h.r.o.n.o.s." text
    showDots: true,     // show six pulsing dots
    size: 320           // ANY size in pixels - scales proportionally
});

// Common sizes:
// 100px - Navigation bars, small icons
// 200px - Medium displays
// 320px - Default, hero sections
// 400px+ - Large hero displays

// Change theme dynamically
const logo = new PhronosLogo('container-id');
logo.setTheme('light');
logo.setSize(400);

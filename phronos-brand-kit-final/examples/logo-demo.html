// Usage

Initialize the logo with JavaScript:

// Basic usage
new PhronosLogo('container-id');

// With options
new PhronosLogo('container-id', {
    theme: 'dark',      // 'dark' or 'light'
    showText: true,     // show ".p.h.r.o.n.o.s." text
    showDots: true,     // show six pulsing dots
    size: 320           // size in pixels
});

// Change theme dynamically
const logo = new PhronosLogo('container-id');
logo.setTheme('light');
logo.setSize(400);
